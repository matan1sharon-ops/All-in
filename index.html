function playBip(count) { if (!state.mode.includes('TARGET')) return; for(let i=0; i { const osc = audioCtx.createOscillator(); const gain = audioCtx.createGain(); // "sine" is a cleaner, more professional beep than "square" osc.type = "sine"; osc.frequency.setValueAtTime(800, audioCtx.currentTime); // Start volume gain.gain.setValueAtTime(0.1, audioCtx.currentTime); // HOLD the volume at 0.1 for 0.4 seconds (this was the missing part!) gain.gain.setValueAtTime(0.1, audioCtx.currentTime + 0.4); // Quick fade out at the very end (0.5s mark) gain.gain.exponentialRampToValueAtTime(0.0001, audioCtx.currentTime + 0.5); osc.connect(gain); gain.connect(audioCtx.destination); osc.start(); osc.stop(audioCtx.currentTime + 0.5); }, i * 800); // Increased gap slightly so 0.5s bips don't overlap } }
